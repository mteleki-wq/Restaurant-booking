                  if (r+1>=exercise.rounds) { setRunning(false); setPhase('done'); return r+1; }
                  return r+1;
                });
              }
              setPhase(np);
              return durations[np];
            }
            return t-1;

          {/* JOURNAL MODAL */}
          {showJournal && (
            <div style={{position:'fixed',inset:0,zIndex:50,display:'flex',alignItems:'center',justifyContent:'center',background:'rgba(20,25,23,0.85)',backdropFilter:'blur(16px)',padding:'1rem'}}>
              <div style={{position:'relative',width:'100%',maxWidth:'36rem',borderRadius:'1.5rem',padding:'2rem',background:'var(--cream)',maxHeight:'90vh',overflowY:'auto'}}>
                <button onClick={()=>setShowJournal(false)} style={{position:'absolute',top:'1.25rem',right:'1.25rem',padding:'0.5rem',borderRadius:'9999px'}}>
                  <SvgIcon name="x" size={20} color="var(--slate)" />
                </button>
                <p style={{fontSize:'0.7rem',letterSpacing:'0.3em',textTransform:'uppercase',marginBottom:'0.5rem',color:'var(--sage-dark)'}}>Day {currentDay} · Reflection</p>
                <h3 className="font-serif" style={{fontSize:'2rem',marginBottom:'1.5rem',color:'var(--ink)'}}>A Quiet Check-In</h3>
                {REFLECTION_QUESTIONS.map((q,i)=>(
                  <div key={i} style={{marginBottom:'1.25rem'}}>
                    <label style={{display:'block',fontSize:'0.875rem',fontStyle:'italic',fontFamily:'Cormorant Garamond,Georgia,serif',marginBottom:'0.5rem',color:'var(--slate)'}}>{q}</label>
                    <textarea value={draft[i]||''} onChange={e=>setDraft({...draft,[i]:e.target.value})}
                      style={{width:'100%',padding:'0.75rem 1rem',borderRadius:'0.75rem',fontSize:'0.875rem',resize:'none',background:'var(--cream-warm)',border:'1px solid rgba(0,0,0,0.08)',color:'var(--ink)',minHeight:'70px',fontFamily:'inherit'}}
                      placeholder="Write freely..." />
                  </div>
                ))}
                <button onClick={saveReflection} style={{width:'100%',marginTop:'0.5rem',padding:'0.75rem 1.25rem',borderRadius:'9999px',fontSize:'0.875rem',letterSpacing:'0.1em',textTransform:'uppercase',background:'var(--ink)',color:'var(--cream)',transition:'transform 0.2s'}} onMouseOver={e=>e.currentTarget.style.transform='scale(1.02)'} onMouseOut={e=>e.currentTarget.style.transform='scale(1)'}>
                  Save Reflection
                </button>
              </div>
            </div>
          )}
        </div>
      );
    }

    ReactDOM.createRoot(document.getElementById('root')).render(<App />);
  </script>
